CREATE TABLE livro (
	id_livro serial,
	cod_livro varchar(8),
	nome varchar(50), 
	editora varchar(50),
	autor varchar(50),
	idioma varchar(30),
	descricao varchar(70),
	preco decimal,
	CONSTRAINT livro_pkey PRIMARY KEY (id_livro),
	CONSTRAINT livro_cod_key UNIQUE (cod_livro)
)
